13  public class GestorAprovisionamiento {
14      
15      Agente agente;
16      
17      public GestorAprovisionamiento() throws Exception {
18          this.agente = Agente.getAgente();
19      }
20      
21      public int aniadirAprovisionamiento(String nombre, String num) throws Exception{
22          Aprovisionamiento aprovisionamiento = new Aprovisionamiento();
23          
24          aprovisionamiento.setNombre(nombre);
25          aprovisionamiento.setCantidad(Integer.parseInt(num));
26          
27          String consulta = "INSERT INTO Aprovisionamiento VALUES(null,'"+ aprovisionamiento.getNombre()+"',"+ aprovisionamiento.getCantidad() +")";
28          
29          int resultado = this.agente.insert(consulta);
30          
31          return resultado;
32      }
