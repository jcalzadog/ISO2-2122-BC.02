14  public class GestorCampania {

18      public GestorCampania() throws Exception {
19          this.agente = Agente.getAgente();
20      }
21      
22      public Vector getCampanias(){
23          
24          Vector<Object> campanias = null;
25          
26          try {
27              campanias = this.agente.select("SELECT * FROM Campania",7);
28             
29          } catch (Exception ex) {
30              System.out.println(ex);
31          }
32          
33          return campanias;
34          
35      }
36      
37      public int anadirCampania(String tipo, String nombre, String fecha, String coste, String tempo, String efec) throws Exception{
38          Campania campania = new Campania();
39          
40          campania.setTipo(tipo);
41          campania.setNombre(nombre);
42          campania.setFecha(fecha);
43          campania.setCoste(Double.parseDouble(coste));
44          campania.setTemporalidad(Integer.parseInt(tempo));
45          campania.setEfectividad(Double.parseDouble(efec));
46          
47          String consulta = "INSERT INTO Campania VALUES (null,'"+ campania.getTipo()+"','"+ campania.getNombre()+"','"+ campania.getFecha() +"',"+ campania.getCoste() +","+ campania.getTemporalidad() +","+ campania.getEfectividad()+")";
48     
49          int resultado = this.agente.insert(consulta);
50          
51          return resultado;
52      }
