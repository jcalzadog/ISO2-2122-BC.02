311     private void btnGuardarActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnGuardarActionPerformed
312 
313         int id;
314         String nombre;
315         String apellidos;
316         int telefono;
317         String fechanac;
318         boolean enfermo;
319         boolean confinado;
320         boolean atencion;
321         boolean sano;
322         boolean cuarentena;
323         boolean vulnerable;
324         boolean vacunado;
325 
326         //COMPROBACIONES DE QUE NO ESTEN VACIOS
327         if (this.txtIdentificador.getText().isEmpty() || this.txtNombre.getText().isEmpty()
328                 || this.txtApellidos.getText().isEmpty() || this.txtTelefono.getText().isEmpty()
329                 || this.txtFechanac.getText().isEmpty() || this.btnGroupAtencion.getSelection() == null
330                 || this.btnGroupCuarentena.getSelection() == null || this.btnGroupSano.getSelection() == null
331                 || this.btnGroupVacunado.getSelection() == null || this.btnGroupVulnerable.getSelection() == null
332                 || this.btngroupConfinados.getSelection() == null) {
333 
334             JOptionPane.showMessageDialog(this, "Estás olvidando algún campo", "Error", JOptionPane.ERROR_MESSAGE);
335         } else {
336             id = Integer.parseInt(this.txtIdentificador.getText());
337             nombre = this.txtNombre.getText();
338             apellidos = this.txtApellidos.getText();
339             telefono = Integer.parseInt(this.txtTelefono.getText());
340             fechanac = this.txtFechanac.getText();
341             enfermo = this.rbSiEnfermo.isSelected();
342             confinado = this.rbSiConfinado.isSelected();
343             atencion = this.rbSiAtencion.isSelected();
344             sano = this.rbSiSano.isSelected();
345             cuarentena = this.rbSiCuarentena.isSelected();
346             vulnerable = this.rbSiVulnerable.isSelected();
347             vacunado = this.rbSiVacunado.isSelected();
348 
349             try {
350                 this.gestorPersona.insertarPersona(id, nombre, apellidos, telefono, fechanac, enfermo, confinado, atencion, sano, cuarentena, vulnerable, vacunado, "no tiene");
351            
352                 this.dispose();
353 
354             } catch (MysqlDataTruncation dt) {
355                 JOptionPane.showMessageDialog(this, "Has introducido la fecha de manera incorrecta.\nIntroduce el formato correcto: YYYY-MM-DD.", "Error", JOptionPane.ERROR_MESSAGE);
356 
357             } catch (SQLIntegrityConstraintViolationException icv) {
358                 JOptionPane.showMessageDialog(this, "Ya hay una persona registrada con ese identificador.", "Error", JOptionPane.ERROR_MESSAGE);
359 
360             } catch (Exception ex) {
361                 Logger.getLogger(IU_AniadirPersonas.class.getName()).log(Level.SEVERE, null, ex);
362                 JOptionPane.showMessageDialog(this, "Has introducido algun campo de manera incorrecta.", "Error", JOptionPane.ERROR_MESSAGE);
363             }
364 
365         }
366 
367     }//GEN-LAST:event_btnGuardarActionPerformed
