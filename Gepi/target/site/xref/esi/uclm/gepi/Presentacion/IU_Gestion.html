184     public void rellenarTabla(Vector tipo, Object[] columna){
185         tabla.getTableHeader().setReorderingAllowed(false);
186         DefaultTableModel modelo = new DefaultTableModel(){
187             @Override
188             public boolean isCellEditable(int row, int col){
189                return false; 
190             }
191         };
192         
193         modelo.setColumnIdentifiers(columna);
194         for(int i = 0; i < tipo.size(); i++){
195             modelo.addRow((Vector) tipo.get(i)); 
196         }
197         //Asigna el modelo a la tabla
198         tabla.setModel(modelo);
199     }
