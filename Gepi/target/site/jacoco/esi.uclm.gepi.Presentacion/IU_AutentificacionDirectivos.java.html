    private void btnAccederActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnAccederActionPerformed

        if(password.equals("admin")){
            IU_Estadisticas estadisticas = new IU_Estadisticas(new javax.swing.JFrame(),true);
            this.setVisible(false);
            estadisticas.setVisible(true);  
        } else {
            JOptionPane.showMessageDialog(null, "La contraseña de Directivos es erronea.", "Autentificación Incorrecta", JOptionPane.WARNING_MESSAGE);
